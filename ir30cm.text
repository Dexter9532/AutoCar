#include <Servo.h>
#include <Wire.h>
#include <OPT3101.h>
#include <Arduino.h>

void setup() {
    Serial.begin(9600);       // starta seriell kommunikation
    pinMode(A1, INPUT);       // A1 är input för IR-sensor
  }
  
  void loop() {
    int irValue = analogRead(A1);   // läs analogt värde från IR-sensor
    Serial.print("IR-värde (A1): ");
    Serial.println(irValue);        // skriv ut värdet
  
    delay(200); // liten paus för att undvika spam
  }
